
local model_sounds = {
	["models/jason278-players/gabe_3.mdl"] = {
		Sound = "http://puu.sh/6z9cf.mp3"
	},
	["models/player/cj.mdl"] = {
		Sound = "http://puu.sh/6F0A8.mp3"
	},
	["models/player/jack_sparrow.mdl"] = {
		Sound = "http://puu.sh/6zeff.mp3" --"http://puu.sh/6zeff.mp3" --"http://puu.sh/6zeff.mp3"
	}
}

hook.Add("Think", "ModelSounds", function()
end)
--[[
	-- http://puu.sh/6z9cf.mp3
	for _,ply in pairs(player.GetAll()) do
		local tbl = model_sounds[ply:GetModel()]
		if IsValid(ply.ModelSound) then
			if (not tbl or not ply:Alive()) then
				ply.ModelSound:Stop()
				ply.ModelSound = nil
			else
				ply.ModelSound:SetPos(ply:EyePos())
			end
		end
		if not IsValid(ply.ModelSound) and tbl and ply:Alive() and not ply.ModelingIntensifies then
			ply.ModelingIntensifies = true
			sound.PlayURL(tbl.Sound, "3d", function(chan)
				ply.ModelSound = chan
				ply.ModelingIntensifies = false
			end)
		end
		--MsgN(ply:GetModel())	
	end
end)
]]
